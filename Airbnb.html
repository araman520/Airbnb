#Cleaning function to clean the price column
def clean(p):
    if type(p) == str:
        p = p.replace("$", "")
        p = p.replace(",", "")
        return float(p)
    else:
        return p 
    
#Clean the price column using the above function
calendar["price"] = calendar["price"].apply(clean)
